Player Data System

A complete player data system with persistence, validation, and replication.

Server Implementation

-- ServerScriptService/PlayerDataSystem.server.lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local DataStoreService = game:GetService("DataStoreService")

local ReplicatedRegistry = require(ReplicatedStorage.ReplicatedRegistry)
local playerDataStore = DataStoreService:GetDataStore("PlayerData")

-- Initialize remotes
local remotes = ReplicatedStorage.ReplicationRemotes
ReplicatedRegistry.server.set_remote_instances(
    remotes.RequestFull,
    remotes.SendChanges
)

-- Default player data template
local function createDefaultData()
    return {
        coins = 100,
        level = 1,
        experience = 0,
        stats = {
            strength = 10,
            speed = 10,
            defense = 10
        },
        inventory = {},
        settings = {
            music = true,
            sfx = true,
            graphics = "Medium"
        },
        lastLogin = os.time()
    }
end

-- Data validation filter
local function createPlayerDataFilter()
    return ReplicatedRegistry.get_filter({
        rate_limit = 10,
        custom = function(sender, path, value)
            if not sender then return true end
            
            -- Block modifications to coins and experience from client
            local field = path[1]
            if field == "coins" or field == "experience" or field == "level" then
                return false
            end
            
            -- Allow settings changes
            if field == "settings" then
                return true
            end
            
            -- Block everything else
            return false
        end
    })
end

-- Load or create player data
local function loadPlayerData(player: Player)
    local key = tostring(player.UserId)
    local success, data = pcall(function()
        return playerDataStore:GetAsync(key)
    end)
    
    if not success or not data then
        data = createDefaultData()
    end
    
    data.lastLogin = os.time()
    
    -- Register with filter
    ReplicatedRegistry.server.register(
        key,
        data,
        createPlayerDataFilter()
    )
    
    -- Send to player
    ReplicatedRegistry.server.to_clients(key, {player})
    
    return data
end

-- Save player data
local function savePlayerData(player: Player)
    local key = tostring(player.UserId)
    local data = ReplicatedRegistry.server.view(key)
    
    pcall(function()
        playerDataStore:SetAsync(key, data)
    end)
end

-- Player joined
Players.PlayerAdded:Connect(function(player)
    loadPlayerData(player)
end)

-- Player leaving
Players.PlayerRemoving:Connect(function(player)
    savePlayerData(player)
end)

-- Periodic auto-save
task.spawn(function()
    while true do
        task.wait(300) -- Save every 5 minutes
        for _, player in Players:GetPlayers() do
            savePlayerData(player)
        end
    end
end)

-- Example: Add coins (from server event)
local function addCoins(player: Player, amount: number)
    local key = tostring(player.UserId)
    local proxy = ReplicatedRegistry.server.view_as_proxy(key)
    
    proxy.incr("coins")(amount)
    proxy.replicate(player)
end

-- Example: Level up system
local function checkLevelUp(player: Player)
    local key = tostring(player.UserId)
    local data = ReplicatedRegistry.server.view(key)
    
    local requiredXP = data.level * 100
    if data.experience >= requiredXP then
        local proxy = ReplicatedRegistry.server.view_as_proxy(key)
        
        proxy.incr("level")(1)
        proxy.incr("experience")(-requiredXP)
        proxy.incr("stats", "strength")(2)
        proxy.incr("stats", "speed")(2)
        proxy.incr("stats", "defense")(2)
        
        proxy.replicate(player)
    end
end

Client Implementation

-- StarterPlayer/StarterPlayerScripts/PlayerDataClient.client.lua
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local ReplicatedRegistry = require(ReplicatedStorage.ReplicatedRegistry)
local player = Players.LocalPlayer

-- Initialize
local remotes = ReplicatedStorage.ReplicationRemotes
ReplicatedRegistry.client.set_remote_instances(
    remotes.RequestFull,
    remotes.SendChanges
)

-- Get player data
local key = tostring(player.UserId)
local playerData = ReplicatedRegistry.client.view(key)

-- Create UI references (example)
local coinsLabel = player.PlayerGui.MainUI.CoinsLabel
local levelLabel = player.PlayerGui.MainUI.LevelLabel

-- Update UI
local function updateUI()
    coinsLabel.Text = "Coins: " .. playerData.coins
    levelLabel.Text = "Level " .. playerData.level
end

updateUI()

-- Listen for changes
ReplicatedRegistry.on_recieve(key, function(sender, tbl, changes)
    updateUI()
    
    -- Handle specific changes
    for _, change in changes do
        if change.p[1] == "level" then
            -- Show level up notification
            print("Level up! Now level", tbl.level)
        end
    end
end)

-- Update settings (client can modify these)
local function updateSettings(settingName: string, value: any)
    local proxy = ReplicatedRegistry.client.view_as_proxy(key)
    proxy.set("settings", settingName)(value)
    proxy.replicate()
end

Real-Time Leaderboard

A leaderboard that updates in real-time for all players.

-- Server
local ReplicatedRegistry = require(ReplicatedStorage.ReplicatedRegistry)

-- Initialize leaderboard
local leaderboardData = {}
ReplicatedRegistry.server.register("global_leaderboard", leaderboardData)

-- Update player score
local function updateScore(player: Player, score: number)
    local proxy = ReplicatedRegistry.server.view_as_proxy("global_leaderboard")
    
    proxy.set(tostring(player.UserId))({
        name = player.Name,
        score = score,
        timestamp = os.time()
    })
    
    -- Broadcast to all players
    proxy.replicate()
end

-- Client
local leaderboardData = ReplicatedRegistry.client.view("global_leaderboard")

local function updateLeaderboardUI()
    -- Convert to sorted array
    local entries = {}
    for userId, data in leaderboardData do
        table.insert(entries, data)
    end
    
    table.sort(entries, function(a, b)
        return a.score > b.score
    end)
    
    -- Update UI
    for i, entry in ipairs(entries) do
        local frame = leaderboardUI:FindFirstChild("Entry" .. i)
        if frame then
            frame.NameLabel.Text = entry.name
            frame.ScoreLabel.Text = tostring(entry.score)
        end
    end
end

updateLeaderboardUI()

ReplicatedRegistry.on_recieve("global_leaderboard", function()
    updateLeaderboardUI()
end)

Game State Management

Manage game rounds, timers, and state across all clients.

-- Server
local ReplicatedRegistry = require(ReplicatedStorage.ReplicatedRegistry)

local gameState = {
    status = "Waiting", -- Waiting, Playing, Ended
    round = 0,
    timeLeft = 0,
    players_alive = {},
    map = "Map1"
}

ReplicatedRegistry.server.register("game_state", gameState)

-- Game loop
local function startRound()
    local proxy = ReplicatedRegistry.server.view_as_proxy("game_state")
    
    proxy.incr("round")(1)
    proxy.set("status")("Playing")
    proxy.set("timeLeft")(180) -- 3 minutes
    
    -- Set alive players
    local alivePlayers = {}
    for _, player in Players:GetPlayers() do
        alivePlayers[tostring(player.UserId)] = true
    end
    proxy.set("players_alive")(alivePlayers)
    
    proxy.replicate()
end

-- Update timer
RunService.Heartbeat:Connect(function(dt)
    local state = ReplicatedRegistry.server.view("game_state")
    
    if state.status == "Playing" then
        local proxy = ReplicatedRegistry.server.view_as_proxy("game_state")
        proxy.incr("timeLeft")(-dt)
        
        -- Only replicate every 0.5 seconds
        if math.floor(state.timeLeft * 2) ~= math.floor((state.timeLeft - dt) * 2) then
            proxy.replicate()
        end
        
        if state.timeLeft <= 0 then
            endRound()
        end
    end
end)

-- Client
local gameState = ReplicatedRegistry.client.view("game_state")

ReplicatedRegistry.on_recieve("game_state", function(sender, tbl, changes)
    -- Update UI based on state
    if tbl.status == "Playing" then
        timerLabel.Text = string.format("%d:%02d", 
            math.floor(tbl.timeLeft / 60),
            math.floor(tbl.timeLeft % 60)
        )
    elseif tbl.status == "Waiting" then
        statusLabel.Text = "Waiting for players..."
    end
    
    -- Check for round changes
    for _, change in changes do
        if change.p[1] == "round" then
            print("Round", change.v, "started!")
        end
    end
end)

Team Communication

Share data within a team.

-- Server
local function createTeamData(teamName: string)
    local teamData = {
        members = {},
        resources = 0,
        upgrades = {},
        base_health = 100
    }
    
    local filter = ReplicatedRegistry.get_filter({
        rate_limit = 5,
        custom = function(sender, path, value)
            if not sender then return true end
            
            -- Check if sender is on this team
            local key = "team_" .. teamName
            local data = ReplicatedRegistry.server.view(key)
            return data.members[tostring(sender.UserId)] ~= nil
        end
    })
    
    ReplicatedRegistry.server.register("team_" .. teamName, teamData, filter)
end

createTeamData("Red")
createTeamData("Blue")

-- Add player to team
local function addToTeam(player: Player, teamName: string)
    local key = "team_" .. teamName
    local proxy = ReplicatedRegistry.server.view_as_proxy(key)
    
    proxy.set("members", tostring(player.UserId))({
        name = player.Name,
        role = "Member"
    })
    
    -- Send to all team members
    local data = ReplicatedRegistry.server.view(key)
    local teamPlayers = {}
    for userId, _ in data.members do
        local p = Players:GetPlayerByUserId(tonumber(userId))
        if p then table.insert(teamPlayers, p) end
    end
    
    ReplicatedRegistry.server.to_clients(key, teamPlayers)
end

-- Client
local myTeam = game.Teams[player.Team.Name]
local teamKey = "team_" .. myTeam.Name
local teamData = ReplicatedRegistry.client.view(teamKey)

-- Team member can update shared data
local function addResources(amount: number)
    local proxy = ReplicatedRegistry.client.view_as_proxy(teamKey)
    proxy.incr("resources")(amount)
    proxy.replicate()
end

Inventory System

A networked inventory with validation.

-- Server
local MAX_INVENTORY_SIZE = 20

local inventoryFilter = function(sender, path, value)
    if not sender then return true end
    
    -- Players can't directly modify their inventory
    -- Must go through server functions
    return false
end

Players.PlayerAdded:Connect(function(player)
    local key = "inventory_" .. player.UserId
    ReplicatedRegistry.server.register(key, {
        items = {},
        capacity = MAX_INVENTORY_SIZE
    }, inventoryFilter)
end)

-- Add item function
local function addItem(player: Player, itemId: string, quantity: number)
    local key = "inventory_" .. player.UserId
    local proxy = ReplicatedRegistry.server.view_as_proxy(key)
    local inventory = proxy.full()
    
    -- Check capacity
    local totalItems = 0
    for _, item in inventory.items do
        totalItems += item.quantity or 1
    end
    
    if totalItems + quantity > inventory.capacity then
        return false, "Inventory full"
    end
    
    -- Add item
    local existingIndex = nil
    for i, item in inventory.items do
        if item.id == itemId then
            existingIndex = i
            break
        end
    end
    
    if existingIndex then
        proxy.incr("items", existingIndex, "quantity")(quantity)
    else
        table.insert(inventory.items, {
            id = itemId,
            quantity = quantity,
            timestamp = os.time()
        })
        proxy.set("items")(inventory.items)
    end
    
    proxy.replicate(player)
    return true
end

-- Client
local key = "inventory_" .. player.UserId
local inventory = ReplicatedRegistry.client.view(key)

local function updateInventoryUI()
    for i, item in inventory.items do
        local slot = inventoryUI:FindFirstChild("Slot" .. i)
        if slot then
            slot.ItemIcon.Image = getItemIcon(item.id)
            slot.Quantity.Text = tostring(item.quantity)
        end
    end
end

updateInventoryUI()

ReplicatedRegistry.on_recieve(key, function()
    updateInventoryUI()
end)

Settings Synchronization

Sync user preferences across sessions.

-- Server
Players.PlayerAdded:Connect(function(player)
    local settings = {
        audio = {
            master = 1.0,
            music = 0.8,
            sfx = 0.9
        },
        graphics = {
            quality = "Medium",
            shadows = true,
            particles = true
        },
        controls = {
            sensitivity = 0.5,
            invertY = false
        }
    }
    
    -- Load from DataStore
    local key = "settings_" .. player.UserId
    local success, savedSettings = pcall(function()
        return settingsStore:GetAsync(player.UserId)
    end)
    
    if success and savedSettings then
        settings = savedSettings
    end
    
    -- Allow client to modify all settings
    local filter = function(sender, path, value)
        if not sender then return true end
        return sender.UserId == player.UserId
    end
    
    ReplicatedRegistry.server.register(key, settings, filter)
    ReplicatedRegistry.server.to_clients(key, {player})
end)

-- Client
local key = "settings_" .. player.UserId
local settings = ReplicatedRegistry.client.view(key)

-- Apply settings
local function applySettings()
    SoundService.Volume = settings.audio.master
    -- Apply other settings...
end

applySettings()

-- Listen for changes
ReplicatedRegistry.on_recieve(key, function()
    applySettings()
end)

-- Update setting
local function updateSetting(category: string, setting: string, value: any)
    local proxy = ReplicatedRegistry.client.view_as_proxy(key)
    proxy.set(category, setting)(value)
    proxy.replicate()
end

-- Usage
updateSetting("audio", "master", 0.7)
updateSetting("graphics", "quality", "High")

Tips for All Examples

  1. Always validate on server: Never trust client data
  2. Use filters: Protect against malicious modifications
  3. Rate limit: Prevent spam and exploits
  4. Batch updates: Group related changes together
  5. Handle disconnects: Save data before players leave
  6. Error handling: Wrap DataStore calls in pcall
  7. UI updates: Only update when necessary
  8. Clear feedback: Show users when data is syncing
