ReplicatedRegistry2 Docs

Getting Started Guide

This guide will walk you through setting up and using ReplicatedRegistry in your Roblox game.

Prerequisites

  • Basic understanding of Roblox Lua scripting
  • Familiarity with RemoteEvents and RemoteFunctions

Step 1: Create Remote Instances

Create a folder in ReplicatedStorage for your remotes:

-- In ReplicatedStorage, create a script that runs once
local remotes = Instance.new("Folder")
remotes.Name = "ReplicationRemotes"
remotes.Parent = game.ReplicatedStorage

local requestFull = Instance.new("RemoteFunction")
requestFull.Name = "RequestFull"
requestFull.Parent = remotes

local sendChanges = Instance.new("RemoteEvent")
sendChanges.Name = "SendChanges"
sendChanges.Parent = remotes

Step 2: Server Setup

Create a server script to register and manage replicated data:

-- ServerScriptService/DataReplication.server.lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")

local ReplicatedRegistry = require(ReplicatedStorage.ReplicatedRegistry)
local remotes = ReplicatedStorage.ReplicationRemotes

-- Initialize the server-side remotes
ReplicatedRegistry.server.set_remote_instances(
    remotes.RequestFull,
    remotes.SendChanges
)

-- Example: Register player data
Players.PlayerAdded:Connect(function(player)
    local playerData = {
        coins = 100,
        level = 1,
        experience = 0,
        inventory = {},
        stats = {
            strength = 10,
            speed = 10
        }
    }
    
    -- Register with the player's UserId as the key
    ReplicatedRegistry.server.register(
        tostring(player.UserId),
        playerData
    )
    
    -- Notify the player's client
    ReplicatedRegistry.server.to_clients(
        tostring(player.UserId),
        {player}
    )
end)

-- Example: Update and replicate data
local function addCoins(player, amount)
    local key = tostring(player.UserId)
    local data = ReplicatedRegistry.server.view(key)
    
    data.coins += amount
    
    -- Send changes to the specific player
    ReplicatedRegistry.server.to_clients(key, {player})
end

-- Clean up when player leaves
Players.PlayerRemoving:Connect(function(player)
    -- Save data, remove from registry, etc.
end)

Step 3: Client Setup

Create a client script to view and interact with replicated data:

-- StarterPlayer/StarterPlayerScripts/DataClient.client.lua
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")

local ReplicatedRegistry = require(ReplicatedStorage.ReplicatedRegistry)
local remotes = ReplicatedStorage.ReplicationRemotes
local player = Players.LocalPlayer

-- Initialize the client-side remotes
ReplicatedRegistry.client.set_remote_instances(
    remotes.RequestFull,
    remotes.SendChanges
)

-- View the player's data
local myKey = tostring(player.UserId)
local myData = ReplicatedRegistry.client.view(myKey)

print("My coins:", myData.coins)
print("My level:", myData.level)

-- Listen for changes
ReplicatedRegistry.on_recieve(myKey, function(sender, tbl, changes)
    print("Data updated!")
    print("Coins:", tbl.coins)
    
    -- Update UI based on changes
    for _, change in changes do
        local path = table.concat(change.p, ".")
        print(string.format("Changed: %s = %s", path, tostring(change.v)))
    end
end)

Step 4: Basic Operations

Modifying Data (Server)

-- Direct modification
local data = ReplicatedRegistry.server.view("player_123")
data.coins = 500
data.stats.strength = 15

-- Send changes to all clients
ReplicatedRegistry.server.to_all_clients("player_123")

-- Or send to specific clients
ReplicatedRegistry.server.to_clients("player_123", {player1, player2})

Using the Proxy Interface (Server)

local proxy = ReplicatedRegistry.server.view_as_proxy("player_123")

-- Set nested values easily
proxy.set("coins")(500)
proxy.set("stats", "strength")(15)

-- Increment values
proxy.incr("level")(1)
proxy.incr("experience")(50)

-- Get values
local currentCoins = proxy.get("coins")

-- Replicate to all clients
proxy.replicate()

-- Or replicate to specific player
proxy.replicate(player)

Modifying and Sending from Client

-- Client-side modification (if server accepts it)
local proxy = ReplicatedRegistry.client.view_as_proxy(myKey)

-- Make local changes
proxy.set("settings", "volume")(0.8)
proxy.set("settings", "graphics")("High")

-- Send to server
proxy.replicate()

Step 5: Adding Filters (Server)

Protect your data with filters:

-- Create a filter that only accepts positive values
local positiveFilter = function(sender, path, value)
    if type(value) == "number" then
        return value >= 0
    end
    return true
end

-- Register with filter
ReplicatedRegistry.server.register(
    "player_123",
    playerData,
    positiveFilter
)

-- Or use built-in filters
local filter = ReplicatedRegistry.get_filter({
    rate_limit = 10, -- Max 10 changes per second
    player_whitelist = {123456, 789012},
    custom = function(sender, path, value)
        -- Only allow changes to "settings"
        return path[1] == "settings"
    end
})

ReplicatedRegistry.server.register("player_123", playerData, filter)

Common Patterns

Pattern 1: Per-Player Data

-- Server
Players.PlayerAdded:Connect(function(player)
    local key = "player_" .. player.UserId
    ReplicatedRegistry.server.register(key, {
        -- player data
    })
end)

-- Client
local key = "player_" .. player.UserId
local data = ReplicatedRegistry.client.view(key)

Pattern 2: Global Game State

-- Server
ReplicatedRegistry.server.register("game_state", {
    round = 1,
    timeLeft = 60,
    gameActive = false
})

-- Update and broadcast
local gameState = ReplicatedRegistry.server.view("game_state")
gameState.round += 1
ReplicatedRegistry.server.to_all_clients("game_state")

-- All clients
local gameState = ReplicatedRegistry.client.view("game_state")
ReplicatedRegistry.on_recieve("game_state", function(_, tbl, changes)
    -- Update UI
end)

Pattern 3: Shared Leaderboard

-- Server
ReplicatedRegistry.server.register("leaderboard", {
    {name = "Player1", score = 1000},
    {name = "Player2", score = 950}
})

-- Client
local leaderboard = ReplicatedRegistry.client.view("leaderboard")
-- Display leaderboard
